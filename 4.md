#include <msp430.h> 

int i;
int flag=0;
int flag2=0;
int desbordamiento=0;
unsigned int contador=0;
int cont_desbord;
char recepcion[100];

void imprimir(char mensaje[], unsigned int longitud);

// 1.-Activar la funcion
// 2.-Habilitar el reset en USCI
// 3.- Fuente
// 4.- Velocidad de TX1RX
// 5.- A trabajar
// 6.- Hab. interrupcion

void main(void)
{
    WDTCTL = WDTPW | WDTHOLD;   // stop watchdog timer

    P1DIR |= BIT0;

    P4SEL |= BIT5 | BIT4;

    P3SEL |= BIT3 | BIT4;

    // ================= UART A1 =================
    UCA1CTL1 |= UCSWRST;
    UCA1CTL1 |= UCSSEL__SMCLK;

    // 115200 baudios
    UCA1BR0 = 8;
    UCA1BR1 = 0;
    UCA1MCTL = UCBRS_6 | UCBRF_0;

    UCA1CTL1 &= ~UCSWRST;
    UCA1IE |= UCRXIE;

    // ================= UART A0 =================
    UCA0CTL1 |= UCSWRST;
    UCA0CTL1 |= UCSSEL__SMCLK;

    // 115200 baudios
    UCA0BR0 = 8;
    UCA0BR1 = 0;
    UCA0MCTL = UCBRS_6 | UCBRF_0;

    UCA0CTL1 &= ~UCSWRST;
    UCA0IE |= UCRXIE;

    // ================= TIMER =================
    TA0CTL |= TASSEL__SMCLK + MC_0;

    TA0CCR0 = 65535;

    TA0CCTL0 |= CCIE;

    __bis_SR_register(GIE);

    while(1)
    {
        if(flag == 1)
        {
            recepcion[27]='\n';

            recepcion[28]='\r';

            imprimir(recepcion,sizeof(recepcion));

            flag = 0;

            contador=0;
        }
    }
}

// ================= FUNCION IMPRIMIR =================
void imprimir(char mensaje[],unsigned int longitud)
{
    for (i=0; i<longitud; i++)
    {
        UCA1TXBUF = mensaje[i];

        __delay_cycles(10000);
    }
}

// ================= UART RX =================
#pragma vector=USCI_A0_VECTOR
__interrupt void PepitoOriginal(void)
{
    if(UCA0RXBUF == '\n' && flag2 == 0)
    {
        flag2=1;

        contador=0;

        TA0CTL = MC__UP;
    }

    if(flag2 == 1)
    {
        if(contador <= 25 && desbordamiento == 0)
        {
            recepcion[contador]= UCA0RXBUF;

            contador++;

            if(UCA0RXBUF == '\n')
            {
                flag = 1;

                contador = 0;

                TA0R = 0;
            }
        }
    }

    if(contador >= 26 || desbordamiento == 1)
    {
        flag = 1;

        flag2 = 0;

        TA0CCTL0 &= ~CCIE;

        TA0CTL = MC_0;

        TA0R = 0;

        contador=0;

        desbordamiento=0;
    }
}

// ================= TIMER =================
#pragma vector=TIMER0_A0_VECTOR
__interrupt void Pepito3(void)
{
    cont_desbord++;

    if(cont_desbord >= 410)
    {
        desbordamiento=1;

        cont_desbord=0;
    }
}
